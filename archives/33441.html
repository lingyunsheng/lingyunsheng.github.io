function getUrlParam(sUrl, sKey) {
  if (!sUrl || typeof sUrl !== "string") {
    return sKey ? "" : {};
  }
  let arr = sUrl.split("?")[1].split("#")[0].split("&");
  for (let i = 0; i < arr.length; i++) {
    //     定义一个对象 键和值对应
    let obj = {};
    let res = [];
    let array = [];
    let objarr = arr[i].split("=");
    array.push(objarr);
    for (let j = 0; j < objarr; j++) {
      if (obj.hashOwnProperty(objarr[j][0])) {
        obj[objarr[j][0]] = array[j][1];
        obj[objarr[j][0]]++;
      }
    }
    if (sKey == "") {
      return obj;
    } else if (obj[sKey]) {
      res.push(obj);
      obj[sKey]++;
      return res;
    }
  }
}

function findKth(a, n, K) {
  // write code here
  const quickSort = (arr) => {
    if (arr.length <= 1) {
      return arr;
    }
    let left = [];
    let right = [];
    let mid = Math.floor(arr.length / 2);
    for (let i = 0; i < arr.length; i++) {
      if (arr[i] < arr[mid]) {
        left.push(arr[i]);
      } else {
        right.push(arr[i]);
      }
    }
    return [...quickSort(left), arr[mid], ...quickSort(right)];
  };
  let arrays = quickSort(a);
  return arrays[n - K];
}

function merge(A, m, B, n) {
  // write code here
  let l1 = m - 1;
  let l2 = n - 1;
  let len = m + n - 1;
  while (l1 >= 0 && l2 >= 0) {
    if (A[l1] > B[l2]) {
      A[len] = A[l1];
      l1--;
    } else {
      A[len] = B[l2];
      l2--;
    }
    len--;
  }
  while (l2 >= 0) {
    A[len--] = B[l2--];
  }
}

function partition(head, x) {
  //  链表为空 返回null
  if(!head) {
    return null;
  }
  //  创建两个空链表 小于x的放x 大于x的按照·原顺序放high链表
  var current = head;
  //  当前的节点
  //  创建小于x的链表和循环的节点
  var lowList = {}
  var lowCur = lowList
  var highList = {}
  var highCUr = highList
  while(current) {
    //  如果current.val < x lowCur的下一个节点就是current
    if(current.val < x) {
      lowCur.next = current
      lowCur = lowCUr.next
    } else {
      highCUr = current
      highCur = highCur.next
    }
    current = current.next
  }
  highCur.next = null;
  lowCur.next = highList.next
  return lowList.next
}

区间内反转链表

将一个链表\ m m 位置到\ n n 位置之间的区间反转，要求时间复杂度 ，空间复杂度 。
例如：
给出的链表为 1\to 2 \to 3 \to 4 \to 5 \to NULL1→2→3→4→5→NULL, ，,
返回 1\to 4\to 3\to 2\to 5\to NULL1→4→3→2→5→NULL.
注意：
给出的 满足以下条件：
1 \leq m \leq n \leq 链表长度1≤m≤n≤链表长度

输入
{1,2,3,4,5},2,4

输出
{1,4,3,2,5}
方法一




/*
 * function ListNode(x){
 *   this.val = x;
 *   this.next = null;
 * }
 */

/**
  * 
  * @param head ListNode类 
  * @param m int整型 
  * @param n int整型 
  * @return ListNode类
  */
function reverseBetween( head ,  m ,  n ) {
    // write code here
    if(!head) return null;
    function ListNode(x) {
        this.val = x
        this.next = null
    }
    let fast = new ListNode(head);
    let slow =new ListNode(head);
    let flag = 1; // 指针每右移一位 flag+1 代表位置
    let slowval = 0
    let fastval = 0
    while(fast && fast.next) {
      flag += 1
      fast = fast.next
      slow = slow.next 
      // 保存m和n所在位置的节点
      if(flag == m) {
        slowval = m
      }
      if(flag == n) {
        fastval = n
      }
      if(slow.next.val == slowval && fast.next.val == fastval) {
        slow = fast.next
        slow.next = fast.next.next
        fast = slow.next
        fast.next = slow.next.next
      }
    }
    return head;
}
module.exports = {
    reverseBetween : reverseBetween
};

function longestValidParentheses( s ) {
    // write code here
    let res = 0;
    //  定义个个栈 s[i]推入
    let stack = [-1];
    for(let i=0; i<s.length; i++){
        if(s[i] === '('){
            stack.push(i);
        } else {
            stack.pop();
            if(stack.length){
                res = Math.max(res, i-stack[stack.length-1]);
            } else {
                stack.push(i);
            }
        }
    }
    return res;
}

完全二叉树的节点

完全二叉树： 一颗深度为k而且有2^k-1个节点的二叉树

满二叉树，所有子树都存在不为空

完全二叉树 最下层的叶子节点几种在树的左部

满二叉树一定是完全二叉树

完全二叉树不一定是满二叉树

完全二叉树判定：

判断一棵树是否是完全二叉树的思路

  • 如果树为空，则直接返回fasle
  • 如果树不为空， 层序遍历二叉树
  • 如果左右子树都不为空 遍历左右子
  • 如果一个节点 左子树为空 右字数不为空 返回false 不是完全二叉树
  • 如果左子树不为空 右字数为空， 或者左右子树都为空 则该树为完全二叉树
