function isSymmetric(root) {
    if(root == null) {
        return true;
    }
    function isSmae(left, right) {
        if(left == null && right == null) {
            return true;
        } else if(left == null || right == null) {
            return false;
        } else if(left.val !==  right.val) {
            return false
        }
        return isSame(left.left, right.right) && isSmae(left.right, right.left);
    }
    return isSymmetric(root.left, root.right);
}

递归法和迭代法的区别

  • 递归法

函数自身调用自身 并且返回结果， 递归函数自身调用自身， 递归是通过重复函数调用实现
重复

  • 迭代法

利用变量的原值推算出变量的一个新值.如果递归是被自己调用自己的话,迭代就是A不停的调用B。

函数内某段代码实现循环

简单地说，递归是重复调用函数自身实现循环。迭代是函数内某段代码实现循环，而迭代与普通循环的区别是：循环代码中参与运算的变量同时是保存结果的变量，当前保存的结果作为下一次循环计算的初始值。

递归中一定有迭代，迭代不一定有递归，递归调用函数，浪费空间，容易造成堆栈的溢出

    sum += i;

输出二叉树的右视图

请根据二叉树的前序遍历，中序遍历恢复二叉树，并打印出二叉树的右视图

输入 [1,2,4,5,3],[4,2,5,1,3]
输出 [1,3,5]
备注 二叉树每个节点的值在区间[1,10000]内，且保证每个节点的值互不相同。

思路： 重建二叉树，层序遍历 根节点[1,] 左子树 [2]

/**
 * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
 * 求二叉树的右视图
 * @param xianxu int整型一维数组 先序遍历    
 * @param zhongxu int整型一维数组 中序遍历
 * @return int整型一维数组
 *    1                    
 *  2   4           2是根节点  4是左子树  5是右子树
 *                       1
 *                     2   3
 *                   4  5 
 * 5 3
 */
function TreeNode(x){
    this.val = x
    this.left = null
    this.right = null
}
//  深度优先遍历 先序遍历 和中序遍历
function dfs(preOrder,inOrder){
    if(preOrder.length === 0 || inOrder.length === 0) return null
    const val = preOrder[0]
    const root  = new TreeNode(val)
    let index = inOrder.indexOf(val)
    //  将先序数组的每一项都赋值val 判断中序遍历是否存在
    root.left = dfs(preOrder.slice(1,1+index),inOrder.slice(0,index))
    root.right = dfs(preOrder.slice(1+index),inOrder.slice(index + 1))
    return root
}

function solve( xianxu ,  zhongxu ) {
    // write code here
    const root = dfs(xianxu,zhongxu)
    const queue = [root],res = []
    while(queue.length){
        let len = queue.length
        for(let i=0;i<len;i++){
            let node = queue.shift()
            if(i == len - 1){
                res.push(node.val)
            }
            if(node.left) queue.push(node.left)
            if(node.right) queue.push(node.right)
        }
    }

    return res
}
module.exports = {
    solve : solve
};

二叉树中是否存在 节点和为 指定sum的路径

给定一个二叉树和一个值\ sum sum，判断是否有从根节点到叶子节点的节点值之和等于\ sum sum 的路径，
例如：
给出如下的二叉树，\ sum=22 sum=22，

输入 {1,2},0

输出 false

输入 {1,2} 3

输出 true

function hasPathSum(root, sum) {
    if(!root) {
        return false;
    }
    //  一条路径一条的去找
    const dfs = (root, sum) => {
        if(root.left == null && root.right == null) {
            //  左右子树都为空则 sum就是根节点的值
            return sum == root.val
        }
        if(root.left && dfs(root.left, sum-root.val)) {
            return true;
        }
        if(root.right && dfs(root.right, sum-root.val)) {
            return true;
        }
        return false;
    }
    return dfs(root, sum)
}

/*
 * function TreeNode(x) {
 *   this.val = x;
 *   this.left = null;
 *   this.right = null;
 * }

/**
  * 
  * @param root TreeNode类 
  * @param sum int整型 
  * @return bool布尔型
  */
function hasPathSum( root ,  sum ) {
    // write code here
    if(!root) {
        return false;
    }
    //  将二叉树全部遍历到一个数组
    let stack = [root]
    let level = 0;
    let res = []
    while(stack.length) {
        let size = stack.length
        let row = []
        while(--size) {
            // 出栈
            let node = stack.shift()
            if(node.left) {
                stack.push(node.left)
            }
            if(node.right) {
                stack.push(node.right)
            }
        }
        res.push([...row]);
    }
    if(!res.length) return;
    // 获得了这个数组 res
    res = res.sort((a,b) => a-b)
    let result = []
    const dfs = (start, curArr, target) => {
        if(target == sum) {
            // 保存路劲数组的和
            result.push([...curArr])
            return true;
        }else if(target < sum) {
            for(let i = start; i<res.length; i++) {
                curArr.push(res[i])
                dfs(i+start, curArr, i+res[i])
                curArr.pop();
                return false;
            }
        }else {
            return false
        }
    }
    dfs(0, [], 0)
}
module.exports = {
    hasPathSum : hasPathSum
};

二叉树的最大深度，

求给定二叉树的最大深度，
最大深度是指树的根结点到最远叶子结点的最长路径上结点的数量。

输入 {1,2}

输出 2

输入 {1,2,3,4,#,#,5}
输出 3

递归法
若是左右节点都存在最后+1

function maxDepth( root ) {

function maxDepth(root) {
    //  根节点不存在
    if(!root) {
        return 0;
    }
    //  分别计算左右子树的最大深度
    return Math.max(maxDepth(root.left), maxDepth(root.right)) + 1;
}

迭代

/*
 * function TreeNode(x) {
 *   this.val = x;
 *   this.left = null;
 *   this.right = null;
 * }
 */

/**
  * 
  * @param root TreeNode类 
  * @return int整型
  */
function maxDepth( root ) {
    // write code here
    if(!root) {
        return 0;
    }
//      添加到栈中
    let stack = [root]
    let count = 0; // 计算节点深度
    while(stack.length) {
        count++
        let size = stack.length
        while(size !== 0) {
            let node = stack.shift()
            if(node.left) {
                stack.push(node.left)
            }
            if(node.right) {
                stack.push(node.right)
            }
            size--;
        }
    }
    return count;
}
module.exports = {
    maxDepth : maxDepth
};

将升序数组转化成平衡二叉搜索树

给出一个升序排序的数组，将其转化为平衡二叉搜索树（BST）.

输入 [-1,0,1,2]
输出 {1,0,2,-1}

平衡二叉搜索树概念:任意节点对应的两颗子树的最大高度差为1,因此称为高度平衡树.查找和插入删除的平均和最欢时间复杂度都是O(logn)

/*
 * function TreeNode(x) {
 *   this.val = x;
 *   this.left = null;
 *   this.right = null;
 * }
 */

/**
  * 
  * @param num int整型一维数组 
  * @return TreeNode类
  */
function sortedArrayToBST( num ) {
    // write code here
    if(num.length==0){
        return null;
    }
    let mid=Math.floor(num.length/2);
    let root=new TreeNode(num[mid]);
    root.left=sortedArrayToBST(num.slice(0,mid));
    root.right=sortedArrayToBST(num.slice(mid+1,num.length));

    return root;
}
module.exports = {
    sortedArrayToBST : sortedArrayToBST
};

字典树

字典树又称为前缀树或者Trie树，是处理字符串常用的数据结构。假设组成所有单词的字符仅是‘a’～‘z’，请实现字典树的结构，并包含以下四个主要的功能。void insert(String word)：添加word，可重复添加；void delete(String word)：删除word，如果word添加过多次，仅删除一次；boolean search(String word)：查询word是否在字典树中出现过(完整的出现过，前缀式不算)；int prefixNumber(String pre)：返回以字符串pre作为前缀的单词数量。现在给定一个m，表示有m次操作，每次操作都为以上四种操作之一。每次操作会给定一个整数op和一个字符串word，op代表一个操作码，如果op为1，则代表添加word，op为2则代表删除word，op为3则代表查询word是否在字典树中，op为4代表返回以word为前缀的单词数量（数据保证不会删除不存在的word）。

对于每次操作，如果op为3时，如果word在字典树中，请输出“YES”，否则输出“NO”；如果op为4时，请输出返回以word为前缀的单词数量，其它情况不输出。

[[“1”,”qwer”],[“1”,”qwe”],[“3”,”qwer”],[“4”,”q”],[“2”,”qwer”],[“3”,”qwer”],[“4”,”q”]]

[“YES”,”2”,”NO”,”1”]

/**
 * 
 * @param operators string字符串二维数组 the ops
 * @return string字符串一维数组
 */
function trieU( operators ) {
    // write code here
    var word=[];
    var res=[];
    for(let i=0;i<operators.length;i++){
        if(operators[i][0]==="1"){
            word.push(operators[i][1]);
        }else if(operators[i][0]==="2"){
            var ind=word.indexOf(operators[i][1]);
            if(ind!==-1){
                word.splice(ind,1);
            }
        }else if(operators[i][0]==="3"){
            var ind2=word.indexOf(operators[i][1]);
            if(ind2!==-1){
                res.push("YES");
            }else{
                res.push("NO");
            }
        }else if(operators[i][0]==="4"){
            var sum=0;
            for(let j=0;j<word.length;j++){
                var wordarr=word[j].split("");
                if(wordarr[0]==operators[i][1]){
                    sum+=1;
                }
            }
            res.push(sum.toString());
        }
    }
    return res;
}
module.exports = {
    trieU : trieU
};
