1.1 包管理 lerna

lerna是包管理工具(多包管理器)， 针对git和npm管理多包存储库的工作流程
安装

npm i lerna -g

初始化

  • init项目初始化
lerna init  // 默认初始化

lerna init --independent // 使用独立版本

lerna init --exact
  • 安装初始化后默认生成的目录
package/  // 默认的包目录

pakage.json // 配置文件

lerna.json // lerna配置

引入包

lerna import package // 本地文件夹的包地址

为所有安装的包安装依赖

lerna add <package>  // 包的名称

// 为目标包安装依赖

lerna add <package> --scope=<target>


// 标记为开发依赖

lerna add <package> --dev

执行命令


lerna run test // 执行包内的test命令

// 执行lerna root包的 dev命令

lerna run --scope=root dev

// 创建一个新包

lerna create package-name

lerna create package-name src

// 在src目录生成包

//  exec执行命令

lerna exec -- rm -fr lib  // 删除包的lib目录

lerna exec --scope root --ls // 查看目录

lerna.json配置

{
  "version": "1.1.3", // 统一包版本， 为 "independent" 时，各个包独立管理版本
  "npmClient": "yarn", // 包命令客户端
  "npmClientArgs": [ "--production" ], // 包命令参数
  "packages": [ // 包管理目录
    "packages/*",
    "apps/*"
  ],
  "command": {  // 命令配置

    "publish": {
      "ignoreChanges": ["ignored-file", "*.md"],
      "message": "chore(release): publish",
      "registry": "https://npm.pkg.github.com"
    },


    "bootstrap": {
      "ignore": "component-*",
      "npmClientArgs": ["--no-package-lock"]
    }


  }

指令

lerna publish

lerna version // lerna notice cli v3.22.1

lerna list

lerna changed

lerna diff

lerna run

lerna init

lerna add

lerna clean

lerna import

lerna create

lerna info

1.2 环境变量 dotenv

dotenv,NPM的一个库，从文件加载环境变量,开发过程中，我们会碰到不同的环境，本地环境，开发环境，测试环境，灰度环境，预发布环境，生产环境，根据不同的环境，我们要配置不同的环境变量， 因此我们是通过dotenv来管理我们的环境变量

安装

npm install --save dotenv

根目录新建一个.env文件,配置不同开发环境的baseurl

#配置开发环境和测试环境以及生产环境
DEV_BASE_URL = 'http://xxx.cn'

TEST_BASE_URL = 'https://xxx.cn'

#客户端请求的url

CLIENT_BASE_URL = 'http://xxx.cn'

#服务端请求的url

SERVER_BASE_URL = 'http://xxx.cn'

#测试环境

TEST2_BASE_URL = 'https://xxx.cn'

在dote.env中引入dotenv


const dotenv = require('dotenv')

const fs = require('fs')

const path = require('path')

// 读取env的环境dotenv.parse

const envConfig = dotenv.parse(fs.readFileSync(path.join(__dirname, '../.env')))

//覆盖系统的环境变量。主要为了解决http_proxy代理问题。
//覆盖服务器上的http_proxy环境变量，否则在压测的时候由于代理性能不足将导致压测数值不达标的问题。

for(let i in envConfig) {
    process.env[i] = envConfig[i]
}

在nuxt.config.js或者vue.dev.config.js中配置

引入dotenv

requir('dotenv').config()

let baseUrl = ''
//  根据不同环境选择
switch(process.env.BASE) {
    case 'dev':
         baseUrl = process.env.DEV_BASE_URL;
         break;
    case 'production':
         baseUrl = process.env.PRO_BASE_URL;
         break;
}
module.exports = {
    env: {
        baseUrl: baseUrl
    }
}
